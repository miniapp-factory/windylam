  const [showResult, setShowResult] = useState(false);

      setShowResult(true);

  const resetQuiz = () => {

      horse: 0,

    setShowResult(false);

  if (showResult) {
    const maxAnimal = Object.entries(scores).reduce((a, b) =>

    )[0];

  }

  const { question, options } = questions[current];
